# Settings

Configuration settings for the agent's behavior

## Example Usage

```typescript
import { Settings } from "@orq-ai/node/models/operations";

let value: Settings = {
  tools: [
    {
      type: "mcp",
      id: "01KA84ND5J0SWQMA2Q8HY5WZZZ",
      toolId: "01KXYZ123456789",
      requiresApproval: false,
    },
  ],
};
```
