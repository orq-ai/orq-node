# DataMessages

## Example Usage

```typescript
import { DataMessages } from "@orq-ai/node/models/operations";

let value: DataMessages = {
  role: "tool",
  content: [
    {
      type: "image_url",
      imageUrl: {
        url: "https://brisk-bob.biz",
      },
    },
  ],
};
```
