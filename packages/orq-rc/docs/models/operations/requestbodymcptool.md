# RequestBodyMCPTool

A tool from a Model Context Protocol (MCP) server that provides standardized access to external capabilities.

## Example Usage

```typescript
import { RequestBodyMCPTool } from "@orq-ai/node/models/operations";

let value: RequestBodyMCPTool = {
  path: "Default",
  key: "<key>",
  description: "ah along runny",
  type: "mcp",
  mcp: {
    serverUrl: "https://prime-backburn.info",
    connectionType: "sse",
  },
};
```
