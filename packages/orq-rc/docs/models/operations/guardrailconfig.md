### `operations.Boolean`

```typescript
const value: operations.Boolean = {
  enabled: false,
  type: "boolean",
  value: false,
};
```

### `operations.NumberT`

```typescript
const value: operations.NumberT = {
  enabled: true,
  type: "number",
  value: 3019.07,
  operator: "gte",
};
```
