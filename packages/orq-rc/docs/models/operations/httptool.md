# HTTPTool

Executes HTTP requests to interact with external APIs and web services using customizable blueprints.

## Example Usage

```typescript
import { HTTPTool } from "@orq-ai/node/models/operations";

let value: HTTPTool = {
  path: "Default",
  key: "<key>",
  description: "midst too hello difficult proofread",
  type: "http",
  http: {
    blueprint: {
      url: "https://darling-laughter.com",
      method: "GET",
    },
  },
};
```
