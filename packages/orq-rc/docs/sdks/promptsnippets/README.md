## create

Create a prompt snippet

### Example Usage

```typescript
import { Orq } from "@orq-ai/node";

const orq = new Orq({
  apiKey: process.env["ORQ_API_KEY"] ?? "",
});

async function run() {
  const result = await orq.promptSnippets.create({
    key: "<key>",
    displayName: "Jed6",
    promptConfig: {
      messages: [
        {
          role: "system",
          content: "<value>",
        },
        {
          role: "system",
          content: [
            {
              type: "image_url",
              imageUrl: {
                url: "https://well-worn-formation.biz",
              },
            },
            {
              type: "text",
              text: "<value>",
            },
            {
              type: "text",
              text: "<value>",
            },
          ],
        },
        {
          role: "assistant",
          content: "<value>",
        },
      ],
    },
    path: "Customer Service/Billing/Refund",
  });

  // Handle the result
  console.log(result);
}

run();
```
